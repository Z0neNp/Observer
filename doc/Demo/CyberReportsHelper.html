def cyber_report_name(cyber_report)
  if cyber_report
    type = cyberReport.class
    creation_date = cyber_report.created_at.strftime('%Y-%m-%dT%H:%M:%S')
    return "#{type} from #{creation_date}"
  end
  'None'
end
